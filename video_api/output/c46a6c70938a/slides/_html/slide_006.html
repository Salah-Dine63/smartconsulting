// Example of a function that might throw an error
function mightFail(shouldFail) {
  if (shouldFail) {
    throw new Error('Sync operation failed!');
  }
  return 'Success';
}

// Demonstrating caught and uncaught errors
try {
  mightFail(true);
} catch (e) {
  console.error('Caught by try...catch:', e.message);
}
